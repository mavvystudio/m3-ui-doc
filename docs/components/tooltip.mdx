# Tooltip

## Example

import Preview from '@site/src/components/Preview';
import Tooltip from '@mavvy/m3-ui/Tooltip';
import Button from '@mavvy/m3-ui/Button';

export function TooltipExample() {
  return (
    <Preview>
      <Tooltip text="Hello World!">
        <Button color="primary" variant="filled" onClick={() => {}}>
          Default Top
        </Button>
      </Tooltip>

      <Tooltip text="Hello World!" yPosition="bottom">
        <Button color="tertiary" variant="filled" onClick={() => {}}>
          Bottom
        </Button>
      </Tooltip>

      <Tooltip text="Hello World!" yPosition="center" xPosition="right">
        <Button color="tertiary" variant="filled" onClick={() => {}}>
          Right Center
        </Button>
      </Tooltip>

      <Tooltip text="Hello World!" yPosition="center" xPosition="left">
        <Button color="tertiary" variant="filled" onClick={() => {}}>
          Left Center
        </Button>
      </Tooltip>
    </Preview>
  );
}

<TooltipExample />

## Props
|prop|type|description|
|----|----|-----------|
|text|string|required. tooltip text|
|children|React.ReactNode|required. tootlip to attach on|
|yPosition|top\|bottom\|center|tooltip position on y axis|
|xPosition|left\|right\|center|tooltip position on x axis|
|className|string|className of the Tooltip component|
|contentClassName|string|className of the Text component container|
|textClassName|string|className of the Text Component|
